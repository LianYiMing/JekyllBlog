def display_menu():
    """展示菜单"""
    print '1. 取五个数的和'
    print '2. 取五个数的平均值'
    print 'x. 退出'
    sel = str(raw_input('输入您的选项：'))
    return sel

def tuple_sum(aTuple):
    """求和"""
    num = 0
    for i in aTuple:
        num += i
    return num

aTuple = (1, 2, 3, 4, 6)
while True :
    sel = display_menu()
    if sel == str('1'):
        print 'num is : %d ' % tuple_sum(aTuple)
    elif sel == str('2'):
        print 'average is : %f ' % (float(tuple_sum(aTuple))/len(aTuple))
    elif sel == str('x'):
        break
    else:
        continue
